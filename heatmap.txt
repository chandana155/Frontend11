    const isNumeric = /^\d+$/.test(q);
    let match;

      // For numbers, find the most exact match
      match = matches.find(a => {
        const name = (a.name || a.area_name || '').toLowerCase();
        const code = (a.code || '').toLowerCase();
        const areaId = (a.area_id || a.id || '').toString().toLowerCase();
        
        // Prioritize exact ID match, then exact number in name/code
        if (areaId === q) return true;
        
        const nameNumbers = name.match(/\d+/g) || [];
        const codeNumbers = code.match(/\d+/g) || [];
        
        return nameNumbers.includes(q) || codeNumbers.includes(q);
      }) || matches[0];
    } else {
      // For non-numeric searches, prioritize exact word matches
      match = matches.find(a => {
        const name = (a.name || a.area_name || '').toLowerCase();
        const code = (a.code || '').toLowerCase();
        const nameWords = name.split(/\s+/);
        const codeWords = code.split(/\s+/);
        
        return nameWords.includes(q) || codeWords.includes(q) || name === q || code === q;
      }) || matches[0];
    }
    
    const hasCoords = Array.isArray(match.coordinates) && match.coordinates.some(pt => typeof pt?.x === 'number' && typeof pt?.y === 'number');
    if (!hasCoords) {
      setHighlightedAreaId(null);
      return;
    }
    setHighlightedAreaId(match.area_id || match.id || null);
    
    // Trigger continuous bounce animation for searched areas
    if (matches.length > 0) {
      setSearchBounceAnimation(true);
    } else {
      setSearchBounceAnimation(false);
    }
  }, [searchTerm, heatmapData.areas]);

            }

            if (zone.cct) {

function MainAreaToggle({ isOn, onClick, isMobile, disabled = false }) {
  const getSize = () => {
    if (window.innerWidth < 600) return { width: 36, height: 18, thumbSize: 14, fontSize: 7 }; // Further reduced
    if (window.innerWidth < 900) return { width: 42, height: 22, thumbSize: 18, fontSize: 8 }; // Further reduced
    return { width: 48, height: 26, thumbSize: 22, fontSize: 9 }; // Further reduced
  };

  const { width, height, thumbSize, fontSize } = getSize();

  return (
    <div
      onClick={disabled ? undefined : onClick}
      style={{
        width,
        height,
        borderRadius: 999,
        background: disabled ? '#ddd' : (isOn ? 'limegreen' : '#bbb'),
        display: 'flex',
        alignItems: 'center',
        cursor: disabled ? 'not-allowed' : 'pointer',
        transition: 'background 0.2s',
        padding: 2,
        position: 'relative',
        minWidth: width,
        flexShrink: 0,
        overflow: 'hidden', // Ensure toggle doesn't overflow
        opacity: disabled ? 0.5 : 1,
      }}
    >
      <div
        style={{
          width: thumbSize,
          height: thumbSize,
          borderRadius: '50%',
          background: '#fff',
          transform: isOn ? `translateX(${width - thumbSize - 4}px)` : 'translateX(0)',
          transition: 'transform 0.2s',
          boxShadow: '0 1px 4px #0001',
          position: 'absolute',
          left: 2,
          top: 2,
        }}
      />
      <span
        style={{
          position: 'absolute',
          left: isOn ? 4 : width - 18, // Adjusted positioning
          color: '#222',
          fontWeight: 700,
          fontSize,
          transition: 'left 0.2s',
          whiteSpace: 'nowrap',
          overflow: 'hidden',
          textOverflow: 'ellipsis',
          maxWidth: width - 10, // Adjusted max width
          lineHeight: 1, // Tighter line height
        }}
      >
        {isOn ? 'ON' : 'OFF'}
      </span>
    </div>
  );
}

function getPolygonBoundingBox(coords) {
  if (!coords.length) return { minX: 0, minY: 0, maxX: 0, maxY: 0, width: 0, height: 0 };
  const xs = coords.map(pt => pt.x);
  const ys = coords.map(pt => pt.y);
  const minX = Math.min(...xs);
  const minY = Math.min(...ys);
  const maxX = Math.max(...xs);
  const maxY = Math.max(...ys);
  return {
    minX, minY, maxX, maxY,
    width: maxX - minX,
    height: maxY - minY,
  };
}

// Helper function to find the largest value in an array
function arrayLargest(arr) {
  if (!arr || arr.length === 0) return 0;
  return Math.max(...arr);
}

function truncateText(text, maxChars) {
  if (text.length <= maxChars) return text;
  return text.slice(0, Math.max(0, maxChars - 1)) + '…';
}

function HeatmapPdfSvgViewer({
  pdfUrl,
  pageDims,
  setPageDims,
  scale,
  setScale,
  hasFit,
  handleFit,
  areas,
  getFill,
  handleAreaClick,
  searchTerm,
  pan,
  setPan,
  searchBounceAnimation,
  isDragging,
  setIsDragging,
  dragStart,
  setDragStart,
  contentBBox,
  containerFitMode,
  highlightedAreaId
}) {
  // Use A4 dimensions as fallback for consistent rendering
  const A4_WIDTH = 794;
  const A4_HEIGHT = 1123;
  
  // Pan/drag functionality
  const handleMouseDown = (e) => {
    if (e.button === 0) { // Left mouse button
      setIsDragging(true);
      setDragStart({ x: e.clientX - pan.x, y: e.clientY - pan.y });
      e.preventDefault();
    }
  };


  const handleMouseMove = (e) => {
    if (isDragging) {
      const newPan = {
        x: e.clientX - dragStart.x,
        y: e.clientY - dragStart.y
      };
      setPan(newPan);
      e.preventDefault();
    }
  };

  const handleMouseUp = (e) => {
    if (isDragging) {
      setIsDragging(false);
      e.preventDefault();
    }
  };

  const handleMouseLeave = (e) => {
    if (isDragging) {
      setIsDragging(false);
    }
  };

  // Scroll-based zoom functionality with mouse-centered zoom
  const handleWheel = (e) => {
    // Define zoom limits
    const MIN_SCALE = 0.1;
    const MAX_SCALE = 5.0;
    
    // Determine zoom direction and factor
    const delta = e.deltaY > 0 ? -1 : 1;
    const zoomFactor = 0.15;
    const newScale = Math.max(MIN_SCALE, Math.min(MAX_SCALE, scale + delta * zoomFactor));
    
    // Get container dimensions and mouse position
    const rect = e.currentTarget.getBoundingClientRect();
    const containerCenterX = rect.width / 2;
    const containerCenterY = rect.height / 2;
    const mouseX = e.clientX - rect.left;
    const mouseY = e.clientY - rect.top;
    
    // Calculate mouse position relative to container center
    const mouseRelativeToCenterX = mouseX - containerCenterX;
    const mouseRelativeToCenterY = mouseY - containerCenterY;
    
    // Calculate the point in the PDF coordinate system that the mouse is pointing at
    // The PDF transform is: translate(-50%, -50%) translate(pan.x, pan.y) scale(scale)
    // So to get the PDF point: (mouse - pan) / scale
    const pdfPointX = (mouseRelativeToCenterX - pan.x) / scale;
    const pdfPointY = (mouseRelativeToCenterY - pan.y) / scale;
    
    // Calculate new pan values to keep the PDF point under the mouse cursor
    // New transform: translate(-50%, -50%) translate(newPan.x, newPan.y) scale(newScale)
    // So: mouse = newPan + (pdfPoint * newScale)
    const newPanX = mouseRelativeToCenterX - pdfPointX * newScale;
    const newPanY = mouseRelativeToCenterY - pdfPointY * newScale;
    
    // Apply the new scale and pan
    setScale(newScale);
    setPan({ x: newPanX, y: newPanY });
  };
  
  return (
    <Box
      sx={{
        width: '100%',
        height: '100%',
        position: 'relative',
        flex: '1 1 auto',
        minHeight: 0,
        minWidth: 0,
        overflow: 'hidden',
        bgcolor: '#fff',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        flexGrow: 1,
        flexShrink: 1,
        cursor: isDragging ? 'grabbing' : 'grab',
        userSelect: 'none',
      }}
      onMouseDown={handleMouseDown}
      onMouseMove={handleMouseMove}
      onMouseUp={handleMouseUp}
      onMouseLeave={handleMouseLeave}
      onWheel={(e) => {
        e.preventDefault();
        handleWheel(e);
      }}
    >
      {/* Centered PDF container with improved positioning */}
      <Box
        sx={{
          position: 'absolute',
          top: '50%',
          left: '50%',
          transform: `translate(-50%, -50%) translate(${pan.x}px, ${pan.y}px) scale(${scale})`,
          width: 'auto',
          height: 'auto',
          maxWidth: 'none', // Remove maxWidth constraint
          maxHeight: 'none', // Remove maxHeight constraint
          display: 'block',
          pointerEvents: isDragging ? 'none' : 'auto', // Disable pointer events while dragging
        }}
      >
        {/* Render full PDF without cropping */}
        <Box>
          {pdfUrl ? (
            <Document file={pdfUrl} key={pdfUrl}>
              <Page
                pageNumber={1}
                width={pageDims ? pageDims.width : A4_WIDTH}
                renderAnnotationLayer={false}
                renderTextLayer={false}
                onLoadSuccess={(page) => {
                  // Capture original PDF dimensions for coordinate scaling
                  const dimensions = { width: page.originalWidth, height: page.originalHeight };
                  setPageDims(dimensions);
                  
                  // Don't call handleFit here - let the useEffect handle it when pageDims changes
                  // This ensures we have the correct dimensions before fitting
                }}
              />
            </Document>
          ) : (
            <CircularProgress />
          )}
          <svg
            width={pageDims ? pageDims.width : A4_WIDTH}
            height={pageDims ? pageDims.height : A4_HEIGHT}
            style={{ position: 'absolute', top: 0, left: 0, zIndex: 2, pointerEvents: 'auto' }}
          >
            {(areas || []).map((area, index) => {
              const coords = (area.coordinates || [])
                .filter(pt => pt && typeof pt.x === 'number' && typeof pt.y === 'number');
              
              // Use coordinates directly - they should already be in the correct PDF coordinate system
              // No scaling needed since we're rendering the PDF at its natural size
              let scaledCoords = coords;
              
              // Use coordinates directly - they should already be in the correct PDF coordinate system
              
              const points = scaledCoords.map((p) => `${p.x},${p.y}`).join(' ');
              const center = scaledCoords.length > 0
                ? { x: scaledCoords.reduce((sum, p) => sum + p.x, 0) / scaledCoords.length, y: scaledCoords.reduce((sum, p) => sum + p.y, 0) / scaledCoords.length }
                : { x: 0, y: 0 };
              const bbox = getPolygonBoundingBox(scaledCoords);

              // Enhanced search highlight - consistent with main search logic
              const q = (searchTerm || "").trim().toLowerCase();
              const searchAreaName = (area.name || area.area_name || "").toLowerCase();
              const areaCode = (area.code || "").toLowerCase();
              const areaId = (area.area_id || area.id || "").toString().toLowerCase();
              
              // Enhanced search matching - same logic as main search
              let isHighlightedSearch = false;
              if (q) {
                const isNumeric = /^\d+$/.test(q);
                
                if (isNumeric) {
                  // For numeric searches, extract all complete numbers from the text
                  const nameNumbers = searchAreaName.match(/\d+/g) || [];
                  const codeNumbers = areaCode.match(/\d+/g) || [];
                  const idNumbers = areaId.match(/\d+/g) || [];
                  
                  // Check if the search number exactly matches any complete number in the area NAME only
                  // Examples:
                  // Search "5" → matches areas with ["5", "15", "25", "54", "154"] (any number containing "5")
                  // Search "54" → matches areas with ["54"] only (not "5" or "4" separately)
                  // Search "543" → matches areas with ["543"] only (not "54" or "5" separately)
                  // NOTE: Only search in area name, NOT in area ID
                  isHighlightedSearch = nameNumbers.includes(q) || 
                                       codeNumbers.includes(q);
                } else {
                  // For non-numeric searches, use word-based matching
                  const nameWords = searchAreaName.split(/\s+/);
                  const codeWords = areaCode.split(/\s+/);
                  
                  isHighlightedSearch = nameWords.some(word => word === q) ||
                                       codeWords.some(word => word === q) ||
                                       searchAreaName.includes(q) || 
                                       areaCode.includes(q) ||
                                       areaId === q;
                }
              }
              const isHighlightedById = highlightedAreaId && ((area.area_id || area.id) === highlightedAreaId);
              const isHighlighted = !!isHighlightedById || !!isHighlightedSearch;

              // Calculate available space - use more space for larger areas
              const areaSize = Math.min(bbox.width, bbox.height);
              // Use more space for larger areas to show full text
              const spaceFactor = areaSize > 50 ? 0.9 : 0.8;
              const availableWidth = bbox.width * spaceFactor;
              const availableHeight = bbox.height * spaceFactor;
              
              // Fixed font size - don't use area size, just use consistent sizing
              const ZOOM_THRESHOLD = 1.0;
              const isZoomedIn = scale > ZOOM_THRESHOLD;
              
              // Fixed font sizes - reduced by 1 point for better visibility
              const fontSize = !isZoomedIn ? 15 : 13; // 15px for abbreviated, 13px for full text
              
              const padding = fontSize * 0.15;
              const lineHeight = fontSize * 1.2;

              // Simple text display: 3-4 letters by default, full names when zoomed
              const wrapText = (text, maxWidth) => {
                if (!text) return [];

                // Use the same zoom threshold as defined above
                const isZoomedIn = scale > ZOOM_THRESHOLD;
                
                if (!isZoomedIn) {
                  // Default view: show 3-4 characters for ALL areas
                  const abbreviatedText = text.substring(0, 4);
                  return [abbreviatedText];
                } else {
                  // Zoomed in: show FULL text - ALWAYS use 2 lines to prevent cutting
                  const words = text.split(' ');
                  
                  // If only one word or very short text, return as is
                  if (words.length <= 1 || text.length <= 8) {
                    return [text];
                  }
                  
                  // For longer text, always split into 2 lines
                  const lines = [];
                  
                  if (words.length === 2) {
                    // Two words - one per line
                    lines.push(words[0]);
                    lines.push(words[1]);
                  } else if (words.length === 3) {
                    // Three words - first word on line 1, rest on line 2
                    lines.push(words[0]);
                    lines.push(words.slice(1).join(' '));
                  } else if (words.length === 4) {
                    // Four words - first two on line 1, last two on line 2
                    lines.push(words.slice(0, 2).join(' '));
                    lines.push(words.slice(2).join(' '));
                  } else {
                    // More than 4 words - split roughly in half
                    const midPoint = Math.ceil(words.length / 2);
                    lines.push(words.slice(0, midPoint).join(' '));
                    lines.push(words.slice(midPoint).join(' '));
                  }
                  
                  return lines;
                }
              };

              // Wrap text to fit within area (up to 3 lines for small areas, 2 lines for larger areas)
              const displayAreaName = area.name || area.area_name || '';
              const finalLines = wrapText(displayAreaName, availableWidth - (padding * 2));

              // Calculate background dimensions - very generous when zoomed in for 2-line text
              const maxLineWidth = Math.max(...finalLines.map(line => line.length * fontSize * 0.4)); // More generous character width for 2-line display
              
              // Set background dimensions based on zoom level and area size
              const isSmallArea = areaSize < 40;
              const backgroundWidth = isZoomedIn 
                ? Math.min(maxLineWidth + (padding * 8), availableWidth * 0.95) // Very generous width for 2-line text
                : Math.min(maxLineWidth + (padding * 2), availableWidth * 0.9);  // Compact when default
              const backgroundHeight = isZoomedIn 
                ? Math.min(finalLines.length * lineHeight + (padding * 6), availableHeight * 0.95) // Very generous height for 2-line text
                : Math.min(finalLines.length * lineHeight + (padding * 1.5), availableHeight * 0.9); // Compact when default

              return (
                <g key={index}>
                  {/* Define clipping path for this area */}
                  <defs>
                    <clipPath id={`clip-${index}`}>
                      <polygon points={points} />
                    </clipPath>
                  </defs>
                  
                  {/* Base polygon - scales with zoom */}
                  <polygon
                    points={points}
                    fill={getFill(area)}
                    stroke={'#000'}
                    strokeWidth={2}
                    vectorEffect="non-scaling-stroke"
                    onClick={() => handleAreaClick(area)}
                    style={{ 
                      cursor: 'pointer', 
                      pointerEvents: 'auto'
                    }}
                  />
                  {center.x && center.y && displayAreaName && finalLines.length > 0 && (
                    <>
                      {/* Multi-line text - consistent size, positioned within the area with strict clipping */}
                      <g 
                        clipPath={`url(#clip-${index})`}
                      >
                      {/* White background rectangle for text */}
                      <rect
                        x={center.x - (backgroundWidth / 2)}
                        y={center.y - (backgroundHeight / 2)}
                        width={backgroundWidth}
                        height={backgroundHeight}
                        fill="white"
                        fillOpacity="0.9"
                        stroke="none"
                        rx="2"
                        ry="2"
                      />
                      {finalLines.map((line, lineIndex) => {
                        // Simple centered text positioning
                        const textX = center.x;
                        const textY = center.y - (finalLines.length * lineHeight / 2) + (lineIndex * lineHeight) + (lineHeight / 2);
                        
                        return (
                        <text
                          key={lineIndex}
                          x={textX}
                          y={textY}
                          textAnchor="middle"
                          dominantBaseline="central"
                          fontSize={fontSize}
                          fill={isHighlighted ? '#b71c1c' : '#000'}
                          stroke="none"
                          fontWeight="600"
                          style={{ 
                            pointerEvents: 'none', 
                            userSelect: 'none', 
                            fontFamily: 'Arial, sans-serif',
                            overflow: 'hidden',
                            textOverflow: 'ellipsis'
                          }}
                        >
                          {line}
                        </text>
                        );
                      })}
                      </g>
                    </>
                  )}
                  {/* Enhanced highlight overlay with red color, thicker border, and continuous bounce animation */}
                  {isHighlighted && (
                    <g
                      style={{ 
                        pointerEvents: 'none',
                        animation: searchBounceAnimation ? 'searchBounce 1.5s ease-in-out infinite' : 'none'
                      }}
                    >
                      <polygon
                        points={points}
                        fill={'none'}
                        stroke={'#ff0000'}
                        strokeWidth={8}
                        vectorEffect="non-scaling-stroke"
                        strokeDasharray="10,5"
                        opacity={1.0}
                      />
                      {/* Additional inner highlight for better visibility */}
                      <polygon
                        points={points}
                        fill={'rgba(255, 0, 0, 0.15)'}
                        stroke={'#ff0000'}
                        strokeWidth={4}
                        vectorEffect="non-scaling-stroke"
                        opacity={0.8}
                      />
                    </g>
                  )}
                </g>
              );
            })}
          </svg>
        </Box>
      </Box>
    </Box>
  );
}

        minHeight: { xs: 36, sm: 38, md: 40 },
      }}>
        <Box sx={{
          display: 'flex',
          flexDirection: 'row',
          alignItems: 'center',
          gap: { xs: 0.3, sm: 0.5, md: 0.8, lg: 1 },
          flex: '1 1 auto',
          minWidth: 0,
          overflow: 'hidden',
        }}>
          {/* Floor Selection */}
          <FormControl
            variant="outlined"
            size="small"
            sx={{
              minWidth: { xs: 110, sm: 125, md: 145, lg: 165 },
              maxWidth: { xs: 120, sm: 140, md: 160, lg: 180 },
              flexShrink: 0,
            }}
          >
            <Select
              value={selectedFloorId || ""}
              onChange={handleFloorChange}
              displayEmpty
              disabled={floorStatus === 'loading'}
              sx={{
                backgroundColor: '#fff',
                color: '#000',
                fontSize: CONTROL_FONT,
                height: CONTROL_HEIGHT
              }}
            >
              <MenuItem value="">Select Floor</MenuItem>
              {getAvailableFloors().map(f => (
                <MenuItem key={f.id} value={f.id}>{f.floor_name}</MenuItem>
              ))}
            </Select>
          </FormControl>

          {/* Search */}
          <Paper
            component="form"
            onSubmit={(e) => e.preventDefault()}
            sx={{
              p: { xs: '1px 2px', sm: '2px 3px' },
              display: 'flex',
              alignItems: 'center',
              width: { xs: 110, sm: 130, md: 200, lg: 250 },
              backgroundColor: 'white',
              height: CONTROL_HEIGHT,
              flexShrink: 0,
              position: 'relative', // Add position relative for absolute positioning of clear icon
            }}
          >
            <IconButton size="small" sx={{ p: { xs: '2px', sm: '4px', md: '6px' } }}>
              <SearchIcon fontSize={isMobile ? 'small' : 'medium'} />
            </IconButton>
            <StyledInputBase
              placeholder="Search Area"
              value={searchTerm}
              onChange={(e) => dispatch(setHeatmapSearchTerm(e.target.value))}
              sx={{ fontSize: CONTROL_FONT }}
              // Disable search if operator doesn't have access to current floor
              disabled={currentUserRole === 'Operator' && selectedFloorId && !canAccessFloor(selectedFloorId)}
            />
            {/* Clear icon - only show when there's text */}
            {searchTerm && (
              <IconButton
                size="small"
                onClick={handleClearSearch}
                sx={{
                  p: { xs: '2px', sm: '4px', md: '6px' },
                  position: 'absolute',
                  right: 0,
                  color: '#666',
                  '&:hover': {
                    color: '#333',
                    backgroundColor: 'rgba(0, 0, 0, 0.04)',
                  },
                }}
              >
                <ClearIcon fontSize={isMobile ? 'small' : 'medium'} />
              </IconButton>
            )}
          </Paper>

          {/* Display Mode Buttons */}
          <Box
            sx={{

              backgroundColor: contentColor, // Use contentColor instead of hardcoded color

              borderRadius: 0.5,
              px: { xs: 0.3, sm: 0.5, md: 0.8, lg: 1 },
              py: { xs: 0.1, sm: 0.2, md: 0.3, lg: 0.4 },
              display: 'flex',
              flexDirection: 'row',
              gap: { xs: 0.1, sm: 0.2, md: 0.3, lg: 0.5 },
              alignItems: 'center',
              boxShadow: 1,
              flexShrink: 0,
            }}
          >
            {['Light', 'Occupancy', 'Energy'].map((mode) => (
              <Button
                key={mode}
                size="small"
                variant="contained"
                disableElevation
                onClick={() => handleDisplayModeChange(mode)}
                sx={{
                  backgroundColor: displayMode === mode ? '#fff' : '#232323',
                  color: displayMode === mode ? '#232323' : '#fff',
                  textTransform: 'none',
                  borderRadius: '50%',
                  height: CONTROL_HEIGHT,
                  minWidth: { xs: 32, sm: 40, md: 50, lg: 58 },
                  px: { xs: 0.4, sm: 0.6, md: 0.8, lg: 1.2 },
                  fontWeight: 600,
                  fontSize: { xs: '0.55rem', sm: '0.65rem', md: '0.75rem', lg: '0.8rem' },
                  boxShadow: 'none',
                  '&:hover': {
                    backgroundColor: displayMode === mode ? '#f5f5f5' : '#444',
                  },
                }}
              >
                {mode}
              </Button>
            ))}
          </Box>
        </Box>


        {/* Action Buttons */}
        <Box sx={{
          display: 'flex',
          justifyContent: 'flex-end',
          flexShrink: 0,
          ml: { xs: 0.3, sm: 0.5, md: 0.8 },
        }}>
          {displayMode === 'Light' && canCreateAreaGroup() && (
            <Button
              size="small"
              variant="contained"
              disableElevation
              onClick={() => navigate('/create-area-model')}
              sx={{
                backgroundColor: '#232323',
                color: '#fff',
                textTransform: 'none',
                borderRadius: 1,
                fontSize: CONTROL_FONT,
                px: { xs: 0.8, sm: 1.1, md: 1.3, lg: 1.6 },
                py: { xs: 0.35, sm: 0.5, md: 0.6, lg: 0.7 },
                minWidth: { xs: 90, sm: 105, md: 115, lg: 125 },
                height: CONTROL_HEIGHT,
              }}
            >
              Create Area Group
            </Button>
          )}

          {displayMode === 'Occupancy' && canModifyDeviceSettings() && (
            <Button
              size="small"
              variant="contained"
              disableElevation
              onClick={() => setShowOccupancyModal(true)}
              sx={{
                backgroundColor: buttonColor,
                color: '#fff',
                textTransform: 'none',
                borderRadius: 1,
                fontSize: CONTROL_FONT,
                px: { xs: 0.8, sm: 1.1, md: 1.3, lg: 1.6 },
                py: { xs: 0.35, sm: 0.5, md: 0.6, lg: 0.7 },
                minWidth: { xs: 90, sm: 105, md: 115, lg: 125 },
                height: CONTROL_HEIGHT,
                '&:hover': { backgroundColor: '#444' }
              }}
            >
              Occupancy Settings
            </Button>
          )}
        </Box>
      </Box>

      {/* Occupancy Modal */}
      <GroupOccupancyModel
        open={showOccupancyModal}
        onClose={() => setShowOccupancyModal(false)}
      />
    </>
  );
};


export default HeatmapControls;
